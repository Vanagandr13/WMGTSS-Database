CREATE TABLE boardTable (
int boardId PRIMARY KEY, // could the module be used as the primary key?
varchar(20) moduleName,
int boardOwner,
);

CREATE TABLE clusterTable (
int clusterId PRIMARY KEY,
int boardId REFERENCES boardTable(boardId),
text displayTitle,
varchar(50) clusterPath,
text description,
);


// how should datafiles relate to data boards and clusters
CREATE TABLE dataFileTable (
int fileID PRIMARY KEY,
int clusterID REFERENCES clusterTable(clusterId),
text fileName NOT NULL,
varchar(20) author NOTNULL,
date uploadDate,
int fileSize,
varchar,
);
